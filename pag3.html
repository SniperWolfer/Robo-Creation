// Pines de motor
const int motorIzq1 = 3;
const int motorIzq2 = 4;
const int motorIzq1 = 5;
const int motorIzq1 = 6;
void setup() {
// Configuración de pines de salida para el controlador L293D
pinMode(motorIzq1, OUTPUT);
pinMode(motorIzq2, OUTPUT);
pinMode(motorDer1, OUTPUT);
pinMode(motorDer2, OUTPUT);
// Iniciar comunicación serial para el módulo Bluetooth
Serial.begin(9600);
}
void loop() {
// Verifica si hay datos disponibles del Bluetooth
if (Serial.available() > 0) {
char command = Serial.read();
switch (command) {
case 'F': // Avanzar
adelante();
break;
case 'B': // Retroceder
atras();
break;
case 'L': // Girar a la izquierda
izquierda();
break;
case 'R': // Girar a la derecha
derecha();
break;
case 'S': // Detenerse
parar();
break;
}
}
}
// Funciones de movimiento
void adelante() {
digitalWrite(motorIzq1, HIGH);
digitalWrite(motorIzq2, LOW);
digitalWrite(motorDer1, HIGH);
digitalWrite(motorDer2, LOW);
}
void atras() {
digitalWrite(motorIzq1, LOW);
digitalWrite(motorIzq2, HIGH);
digitalWrite(motorDer1, LOW);
digitalWrite(motorDer2, HIGH);
}
void izquierda() {
digitalWrite(motorIzq1, LOW);
digitalWrite(motorIzq2, HIGH);
digitalWrite(motorDer1, HIGH);
digitalWrite(motorDer2, LOW);
}
void derecha() {
digitalWrite(motorIzq1, HIGH);
digitalWrite(motorIzq2, LOW);
digitalWrite(motorDer1, LOW);
digitalWrite(motorDer2, HIGH);
}
void parar() {
digitalWrite(motorIzq1, LOW);
digitalWrite(motorIzq2, LOW);
digitalWrite(motorDer1, LOW);
digitalWrite(motorDer2, LOW);
}
